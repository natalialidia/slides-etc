\NewDocumentCommand{\Xvtx}{o m}{
  % #1 = optional hat (if present)
  % #2 = inner exponent (uvw)
  \IfNoValueTF{#1}
    {x^{{#2}}} % No accent
    {%
      \IfEqCase{#1}{
        {hat}{\hat{x}^{{#2}}}
        {bar}{\bar{x}^{{#2}}}
        {tilde}{\tilde{x}^{{#2}}}
        {dot}{\dot{x}^{{#2}}}
        {dots}{\ddot{x}^{{#2}}}
      }[\PackageError{nestedexp}{Unknown accent '#1'}{}]%
    }
}

        {dots}{\ddot{y}^{{#2}}}
      }[\PackageError{nestedexp}{Unknown accent '#1'}{}]%
    }
}
